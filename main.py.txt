def get_user(user_id):
    conn = sqlite3.connect('psychology_bot.db')
    c = conn.cursor()
    c.execute("SELECT * FROM users WHERE user_id = ?", (user_id,))
    user = c.fetchone()
    conn.close()
    return user

def create_user(user):
    conn = sqlite3.connect('psychology_bot.db')
    c = conn.cursor()
    c.execute('''INSERT OR IGNORE INTO users 
                 (user_id, username, first_name, last_name, free_messages_left) 
                 VALUES (?, ?, ?, ?, ?)''',
              (user.id, user.username, user.first_name, user.last_name, 5))
    conn.commit()
    conn.close()

def update_user_premium(user_id, months=1):
    conn = sqlite3.connect('psychology_bot.db')
    c = conn.cursor()
    premium_until = datetime.now() + timedelta(days=30*months)
    
    c.execute('''UPDATE users SET 
                 is_premium = 1, 
                 premium_until = ?,
                 free_messages_left = 0 
                 WHERE user_id = ?''',
              (premium_until, user_id))
    conn.commit()
    conn.close()

def use_free_message(user_id):
    conn = sqlite3.connect('psychology_bot.db')
    c = conn.cursor()
    c.execute("UPDATE users SET free_messages_left = free_messages_left - 1 WHERE user_id = ?", (user_id,))
    conn.commit()
    conn.close()

def save_conversation(user_id, user_message, bot_response, is_premium):
    conn = sqlite3.connect('psychology_bot.db')
    c = conn.cursor()
    c.execute('''INSERT INTO conversations (user_id, message_text, bot_response, is_premium) 
                 VALUES (?, ?, ?, ?)''', (user_id, user_message, bot_response, is_premium))
    conn.commit()
    conn.close()
